export default { async fetch(request, env) { const CORS = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "POST, OPTIONS", "Access-Control-Allow-Headers": "Content-Type", }; // Handle preflight if (request.method === "OPTIONS") { return new Response(null, { status: 204, headers: CORS }); } // Only POST allowed if (request.method !== "POST") { return new Response(JSON.stringify({ error: "Method not allowed" }), { status: 405, headers: { ...CORS, "Content-Type": "application/json" }, }); } // Parse incoming body let prompt, system; try { const body = await request.json(); prompt = body.prompt; system = body.system; } catch { return new Response(JSON.stringify({ error: "Invalid JSON body" }), { status: 400, headers: { ...CORS, "Content-Type": "application/json" }, }); } if (!prompt) { return new Response(JSON.stringify({ error: "Missing 'prompt' field" }), { status: 400, headers: { ...CORS, "Content-Type": "application/json" }, }); } // Build Anthropic request const anthropicBody = { model: "claude-sonnet-4-20250514", max_tokens: 1000, messages: [{ role: "user", content: prompt }], }; if (system) anthropicBody.system = system; // Call Anthropic API let anthropicRes; try { anthropicRes = await fetch("https://api.anthropic.com/v1/messages", { method: "POST", headers: { "Content-Type": "application/json", "x-api-key": env.ANTHROPIC_API_KEY, "anthropic-version": "2023-06-01", }, body: JSON.stringify(anthropicBody), }); } catch (err) { return new Response(JSON.stringify({ error: "Upstream fetch failed", detail: err.message }), { status: 502, headers: { ...CORS, "Content-Type": "application/json" }, }); } // Forward response const data = await anthropicRes.json(); if (!anthropicRes.ok) { return new Response(JSON.stringify({ error: "Anthropic error", detail: data }), { status: anthropicRes.status, headers: { ...CORS, "Content-Type": "application/json" }, }); } // Return clean JSON return new Response(JSON.stringify(data), { status: 200, headers: { ...CORS, "Content-Type": "application/json" }, }); }, };
